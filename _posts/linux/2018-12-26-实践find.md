```shell
# 查找src目录及子目录下以.class结尾的文件
$ find src -name *.class
# 查找src目录及子目录下以.class结尾的文件，然后删除他们
$ find src -name *.class -exec rm {} \;
# 查找src目录及子目录下以.class结尾的文件，然后删除他们，删除前会请求确认
$ find src -name *.class -ok rm {} \;
```
